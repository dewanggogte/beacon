## Phase 1: Foundation (Tasks 1-5)

These tasks have no dashboard dependencies. Task 1 must complete first (DB schema). Tasks 2-4 depend on Task 1. Task 5 (dark mode) is fully independent.

**IMPORTANT — Codebase conventions (applies to ALL tasks):**
- **Imports:** Use `import { db, schema } from '@screener/shared';` — NOT deep path imports. Access tables as `schema.analysisResults`, `schema.companies`, etc.
- **Query patterns:** Use `schema.tableName` prefix consistently in all Drizzle queries (matching existing `queries.ts` pattern).
- **LLM client:** Use `createLlmClient()` from `../llm/llm-client.js`, then call `client.generate(systemPrompt, userPrompt, options)`. There is no `callLlm` function. Model names are full strings like `'claude-haiku-4-5'`.
- **Array filters:** Use Drizzle's `inArray()` operator, not raw SQL `ANY()`.
- **D/E metric:** Debt-to-equity is NOT a top-level column on `company_snapshots` — it lives in the `ratios` JSONB. Extract it server-side where needed, or mark as unavailable with a TODO.

```typescript
import { eq, and } from 'drizzle-orm';

export async function saveAnalysisHistory(scrapeRunId: number): Promise<number> {
  // Batch fetch: JOIN analysis_results with company_snapshots in one query
  const rows = await db.select({
    companyId: schema.analysisResults.companyId,
    finalScore: schema.analysisResults.finalScore,
    classification: schema.analysisResults.classification,
    convictionLevel: schema.analysisResults.convictionLevel,
    classificationSource: schema.analysisResults.classificationSource,
    valuationScore: schema.analysisResults.valuationScore,
    qualityScore: schema.analysisResults.qualityScore,
    governanceScore: schema.analysisResults.governanceScore,
    safetyScore: schema.analysisResults.safetyScore,
    momentumScore: schema.analysisResults.momentumScore,
    buffettScore: schema.analysisResults.buffettScore,
    grahamScore: schema.analysisResults.grahamScore,
    pabraiRiskScore: schema.analysisResults.pabraiRiskScore,
    lynchCategoryScore: schema.analysisResults.lynchCategoryScore,
    lynchClassification: schema.analysisResults.lynchClassification,
    disqualified: schema.analysisResults.disqualified,
    disqualificationReasons: schema.analysisResults.disqualificationReasons,
    piotroskiFScore: schema.analysisResults.piotroskiFScore,
    altmanZScore: schema.analysisResults.altmanZScore,
    beneishMScore: schema.analysisResults.beneishMScore,
    // Snapshot metrics
    stockPe: schema.companySnapshots.stockPe,
    roce: schema.companySnapshots.roce,
    roe: schema.companySnapshots.roe,
    marketCap: schema.companySnapshots.marketCap,
    dividendYield: schema.companySnapshots.dividendYield,
  })
    .from(schema.analysisResults)
    .leftJoin(schema.companySnapshots, and(
      eq(schema.companySnapshots.companyId, schema.analysisResults.companyId),
      eq(schema.companySnapshots.scrapeRunId, scrapeRunId)
    ))
    .where(eq(schema.analysisResults.scrapeRunId, scrapeRunId));

  if (rows.length === 0) {
    console.log('[history] No analysis results for run', scrapeRunId);
    return 0;
  }

  // Batch insert with onConflictDoUpdate
  const values = rows.map(r => ({
    companyId: r.companyId,
    scrapeRunId: scrapeRunId,
    finalScore: r.finalScore,
    classification: r.classification,
    convictionLevel: r.convictionLevel,
    classificationSource: r.classificationSource,
    dimensionScores: {
      valuation: r.valuationScore ? Number(r.valuationScore) : null,
      quality: r.qualityScore ? Number(r.qualityScore) : null,
      governance: r.governanceScore ? Number(r.governanceScore) : null,
      safety: r.safetyScore ? Number(r.safetyScore) : null,
      momentum: r.momentumScore ? Number(r.momentumScore) : null,
    },
    frameworkScores: {
      buffett: r.buffettScore ? Number(r.buffettScore) : null,
      graham: r.grahamScore ? Number(r.grahamScore) : null,
      pabrai: r.pabraiRiskScore ? Number(r.pabraiRiskScore) : null,
      lynch: r.lynchCategoryScore ? Number(r.lynchCategoryScore) : null,
    },
    lynchCategory: r.lynchClassification,
    disqualified: r.disqualified,
    disqualificationReasons: r.disqualificationReasons,
    keyMetrics: {
      pe: r.stockPe ? Number(r.stockPe) : null,
      roce: r.roce ? Number(r.roce) : null,
      roe: r.roe ? Number(r.roe) : null,
      de: null, // TODO: extract from ratios JSONB
      market_cap: r.marketCap ? Number(r.marketCap) : null,
      piotroski: r.piotroskiFScore,
      altman: r.altmanZScore ? Number(r.altmanZScore) : null,
      beneish: r.beneishMScore ? Number(r.beneishMScore) : null,
      dividend_yield: r.dividendYield ? Number(r.dividendYield) : null,
    },
  }));

  // Drizzle supports batch insert — insert all rows in one query
  // Use chunks of 500 to avoid query size limits
  const CHUNK_SIZE = 500;
  for (let i = 0; i < values.length; i += CHUNK_SIZE) {
    const chunk = values.slice(i, i + CHUNK_SIZE);
    await db.insert(schema.analysisHistory).values(chunk).onConflictDoUpdate({
      target: [schema.analysisHistory.companyId, schema.analysisHistory.scrapeRunId],
      set: {
        finalScore: schema.analysisHistory.finalScore, // Will be set from excluded row
        classification: schema.analysisHistory.classification,
        // Note: for full upsert, Drizzle uses the values from the INSERT for the SET
        // The implementer should verify the exact onConflictDoUpdate behavior with batch inserts
      },
    });
  }

  console.log(`[history] Saved ${values.length} analysis history records for run ${scrapeRunId}`);
  return values.length;
}
```

```typescript
import { db, schema } from '@screener/shared';
import { eq, desc, sql, and } from 'drizzle-orm';
import { createLlmClient } from '../llm/llm-client.js';

export async function generateMarketCommentary(scrapeRunId: number): Promise<string | null> {
  console.log('[market-commentary] Generating market commentary for run', scrapeRunId);

  // 1. Classification distribution
  const classificationCounts = await db.select({
    classification: schema.analysisResults.classification,
    count: sql<number>`count(*)::int`,
  })
    .from(schema.analysisResults)
    .where(eq(schema.analysisResults.scrapeRunId, scrapeRunId))
    .groupBy(schema.analysisResults.classification);

  // 2. Sector averages
  const sectorAverages = await db.select({
    sector: schema.companies.sector,
    avgScore: sql<number>`round(avg(${schema.analysisResults.finalScore})::numeric, 1)`,
    count: sql<number>`count(*)::int`,
  })
    .from(schema.analysisResults)
    .innerJoin(schema.companies, eq(schema.analysisResults.companyId, schema.companies.id))
    .where(eq(schema.analysisResults.scrapeRunId, scrapeRunId))
    .groupBy(schema.companies.sector)
    .orderBy(sql`avg(${schema.analysisResults.finalScore}) desc`);

  // 3. Notable movers (scoreChange)
  const movers = await db.select({
    name: schema.companies.name,
    scoreChange: schema.analysisResults.scoreChange,
    classificationChange: schema.analysisResults.classificationChange,
    classification: schema.analysisResults.classification,
  })
    .from(schema.analysisResults)
    .innerJoin(schema.companies, eq(schema.analysisResults.companyId, schema.companies.id))
    .where(and(
      eq(schema.analysisResults.scrapeRunId, scrapeRunId),
      sql`abs(${schema.analysisResults.scoreChange}) >= 10`
    ))
    .orderBy(sql`abs(${schema.analysisResults.scoreChange}) desc`)
    .limit(20);

  // 4. Macro regime (latest)
  const [macro] = await db.select()
    .from(schema.macroSnapshots)
    .orderBy(desc(schema.macroSnapshots.snapshotDate))
    .limit(1);

  // 5. Total companies
  const total = classificationCounts.reduce((sum, c) => sum + c.count, 0);

  // Build prompt
  const classDistStr = classificationCounts.map(c => `${c.classification}: ${c.count}`).join(', ');
  const sectorStr = sectorAverages.slice(0, 10).map(s => `${s.sector}: ${s.avgScore} avg (${s.count} companies)`).join('\n  ');
  const moversStr = movers.length > 0
    ? movers.map(m => `${m.name}: ${Number(m.scoreChange) > 0 ? '+' : ''}${m.scoreChange} pts${m.classificationChange ? ` (${m.classificationChange})` : ''}`).join('\n  ')
    : 'No significant movers (first run or stable market)';
  const macroStr = macro
    ? `Regime: ${macro.regime}, Repo rate: ${macro.repoRate}%, CPI: ${macro.cpi}%, GDP growth: ${macro.gdpGrowth}%`
    : 'No macro data available';

  const prompt = `You are a market analyst writing a brief overview for Indian stock investors.

Data provided:
- Total companies analyzed: ${total}
- Classification distribution: ${classDistStr}
- Top sectors by average score:
  ${sectorStr}
- Notable movers since last analysis:
  ${moversStr}
- Current macro regime: ${macroStr}

Write 3-5 paragraphs covering:
1. Overall market quality (what % scored well/poorly)
2. Sector themes (which sectors stand out, positively or negatively)
3. Notable shifts from last analysis (if available)
4. Macro context and its implications (if macro data available)

Tone: Informative, measured, no hype. Written for investors, not traders. Do not use bullet points or headers — write in flowing prose paragraphs.`;

  try {
    const client = createLlmClient();
    const systemPrompt = 'You are a market analyst writing brief overviews for Indian stock investors. Tone: informative, measured, no hype.';
    const commentary = await client.generate(systemPrompt, prompt, {
      model: 'claude-haiku-4-5',
      maxTokens: 2048,
    });

    // Save to scrape_runs
    await db.update(schema.scrapeRuns)
      .set({ marketCommentary: commentary })
      .where(eq(schema.scrapeRuns.id, scrapeRunId));

    console.log(`[market-commentary] Saved commentary (${commentary.length} chars) for run ${scrapeRunId}`);
    return commentary;
  } catch (error) {
    console.error('[market-commentary] Failed to generate commentary:', error);
    // Non-fatal — pipeline continues without commentary
    return null;
  }
}
```

Create `packages/dashboard/src/components/theme-provider.tsx`:

```tsx
'use client';

import { createContext, useContext, useEffect, useState, type ReactNode } from 'react';

type Theme = 'light' | 'dark' | 'system';

interface ThemeContextType {
  theme: Theme;
  setTheme: (theme: Theme) => void;
  resolvedTheme: 'light' | 'dark';
}

const ThemeContext = createContext<ThemeContextType>({
  theme: 'system',
  setTheme: () => {},
  resolvedTheme: 'light',
});

export function useTheme() {
  return useContext(ThemeContext);
}

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [theme, setThemeState] = useState<Theme>('system');
  const [resolvedTheme, setResolvedTheme] = useState<'light' | 'dark'>('light');

  useEffect(() => {
    const stored = localStorage.getItem('theme') as Theme | null;
    if (stored) setThemeState(stored);
  }, []);

  useEffect(() => {
    const resolved = theme === 'system'
      ? (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light')
      : theme;

    setResolvedTheme(resolved);
    document.documentElement.classList.toggle('dark', resolved === 'dark');
  }, [theme]);

  // Listen for system preference changes
  useEffect(() => {
    if (theme !== 'system') return;
    const mq = window.matchMedia('(prefers-color-scheme: dark)');
    const handler = (e: MediaQueryListEvent) => {
      setResolvedTheme(e.matches ? 'dark' : 'light');
      document.documentElement.classList.toggle('dark', e.matches);
    };
    mq.addEventListener('change', handler);
    return () => mq.removeEventListener('change', handler);
  }, [theme]);

  const setTheme = (t: Theme) => {
    setThemeState(t);
    localStorage.setItem('theme', t);
  };

  return (
    <ThemeContext.Provider value={{ theme, setTheme, resolvedTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}
```

```tsx
'use client';

import { createContext, useContext, useState, useEffect, type ReactNode } from 'react';

interface WatchlistContextType {
  watchlist: string[]; // screener codes
  toggle: (code: string) => void;
  isWatched: (code: string) => boolean;
  clear: () => void;
}

const WatchlistContext = createContext<WatchlistContextType>({
  watchlist: [],
  toggle: () => {},
  isWatched: () => false,
  clear: () => {},
});

export function useWatchlist() {
  return useContext(WatchlistContext);
}

const MAX_WATCHLIST = 20;
const STORAGE_KEY = 'beacon-watchlist';

export function WatchlistProvider({ children }: { children: ReactNode }) {
  const [watchlist, setWatchlist] = useState<string[]>([]);

  useEffect(() => {
    const stored = localStorage.getItem(STORAGE_KEY);
    if (stored) {
      try { setWatchlist(JSON.parse(stored)); } catch {}
    }
  }, []);

  const persist = (codes: string[]) => {
    setWatchlist(codes);
    localStorage.setItem(STORAGE_KEY, JSON.stringify(codes));
  };

  const toggle = (code: string) => {
    if (watchlist.includes(code)) {
      persist(watchlist.filter(c => c !== code));
    } else {
      if (watchlist.length >= MAX_WATCHLIST) {
        alert(`Watchlist is limited to ${MAX_WATCHLIST} companies. Remove one before adding more.`);
        return;
      }
      persist([...watchlist, code]);
    }
  };

  const isWatched = (code: string) => watchlist.includes(code);
  const clear = () => persist([]);

  return (
    <WatchlistContext.Provider value={{ watchlist, toggle, isWatched, clear }}>
      {children}
    </WatchlistContext.Provider>
  );
}
```

```tsx
import { metricDefinitions, formatMetric } from '../lib/metric-definitions.js';
import { Tooltip, InfoIcon } from './tooltip.js';

interface MetricValue {
  key: string;
  value: number | null;
  sectorMedian?: number | null;
}

export function MetricStrip({ metrics }: { metrics: MetricValue[] }) {
  return (
    <div className="flex gap-px bg-border dark:bg-dark-border rounded-lg overflow-hidden">
      {metrics.map(m => {
        const def = metricDefinitions[m.key];
        const contextLine = getContextLine(m, def);
        return (
          <div key={m.key} className="flex-1 bg-bg-card dark:bg-dark-bg-card p-3 text-center min-w-0">
            <div className="text-[11px] text-text-muted dark:text-dark-text-muted uppercase flex items-center justify-center gap-0.5">
              {def?.fullName?.split(' ')[0] || m.key}
              {def && (
                <Tooltip content={<><strong>{def.fullName}</strong><br />{def.explanation}<br /><em>{def.goodRange}</em></>}>
                  <InfoIcon />
                </Tooltip>
              )}
            </div>
            <div className="text-lg font-semibold text-text-primary dark:text-dark-text-primary">
              {formatMetric(m.key, m.value)}
            </div>
            {contextLine && (
              <div className={`text-[10px] ${contextLine.color}`}>{contextLine.text}</div>
            )}
          </div>
        );
      })}
    </div>
  );
}

function getContextLine(m: MetricValue, def: MetricDefinition | undefined): { text: string; color: string } | null {
  if (!def || m.value == null || m.sectorMedian == null) return null;

  const diff = m.value - m.sectorMedian;
  const pctDiff = Math.abs(diff / m.sectorMedian) * 100;

  if (pctDiff < 10) {
    return { text: `≈ sector ${formatMetric(m.key, m.sectorMedian)}`, color: 'text-accent-amber' };
  }

  const isBetter = def.higherIsBetter ? diff > 0 : diff < 0;
  const direction = diff > 0 ? '▲' : '▼';

  return {
    text: `${direction} vs sector ${formatMetric(m.key, m.sectorMedian)}`,
    color: isBetter ? 'text-accent-green' : 'text-accent-red',
  };
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getWatchlistCompanies, getLatestRunId } from '../../../lib/queries.js';

export async function POST(request: NextRequest) {
  const { codes } = await request.json();
  const runId = await getLatestRunId();
  if (!runId || !codes?.length) return NextResponse.json([]);
  const data = await getWatchlistCompanies(codes, runId);
  return NextResponse.json(data);
}
```
